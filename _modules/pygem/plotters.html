from mpl_toolkits import mplot3d
import matplotlib.pyplot as plt
from stl import mesh
import pygem.utils as utils



[docs]def plot_opts(save_fig=True, fig_type='.png', fig_dir='figures'):
	"""
	This method sets some options for the plots.

	:param bool save_fig: flag to save or not the image into a directory called
		figures. Default is True.
	:param str fig_type: the file extension of the file to save. Default is '.png'
	
	:return: opts: the chosen options. The keys in the dictionary are `fig_type`,
		`save_fig`, and `font`. The `font` is a dictionary that sets the font
		properties of the figures.
	:rtype: dict
	"""
	# make figs directory
	if save_fig:
		if not os.path.isdir(fig_dir):
			os.mkdir(fig_dir)

	# set plot fonts
	font = {'family' : 'arial',
			'weight' : 'normal',
			'size' : 14}

	opts = {'fig_type' : fig_type,
			'save_fig' : save_fig,
			'fig_dir' : fig_dir,
			'font' : font}

	return opts




[docs]def plot_stl(filename, output_name=None, opts=None):
	"""
	Method to plot an stl file.

	:param str filename: the stl filename you want to plot.
	:param str output_name: filename of the output file. If None it takes
		the filename and changes the extension accordingly with the opts.
	:param dict opts: dictionary with the options for the plot. It is set
		by :func:`plot_opts`
	"""
	utils.check_variable_type('filename', filename, str)

	if opts is None:
		opts = plot_opts()

	if output_name is None:
		output_name = filename.split('.')[0]
	else:
		utils.check_variable_type('output_name', output_name, str)

	# Create a new plot
	figure = plt.figure()
	axes = mplot3d.Axes3D(figure)
	plt.rc('font', **opts['font'])

	# Load the STL files and add the vectors to the plot
	stl_mesh = mesh.Mesh.from_file(filename)
	axes.add_collection3d(mplot3d.art3d.Poly3DCollection(stl_mesh.vectors))
	# Get the limits of the axis and center the geometry
	max_dim = np.array([np.max(stl_mesh.vectors[:,:,0]), \
		np.max(stl_mesh.vectors[:,:,1]), \
		np.max(stl_mesh.vectors[:,:,2])])
	min_dim = np.array([np.min(stl_mesh.vectors[:,:,0]), \
		np.min(stl_mesh.vectors[:,:,1]), \
		np.min(stl_mesh.vectors[:,:,2])])
	max_lenght = np.max(max_dim - min_dim)
	axes.set_xlim(
		-.6 * max_lenght + (max_dim[0] + min_dim[0]) / 2,
		.6 * max_lenght + (max_dim[0] + min_dim[0]) / 2
	)
	axes.set_ylim(
		-.6 * max_lenght + (max_dim[1] + min_dim[1]) / 2,
		.6 * max_lenght + (max_dim[1] + min_dim[1]) / 2
	)
	axes.set_zlim(
		-.6 * max_lenght + (max_dim[2] + min_dim[2]) / 2,
		.6 * max_lenght + (max_dim[2] + min_dim[2]) / 2
	)

	if opts['save_fig']:
		outfile = opts['fig_dir'] + '/' + output_name + opts['fig_type']
		figure.savefig(outfile)
	else:
		plt.show()
